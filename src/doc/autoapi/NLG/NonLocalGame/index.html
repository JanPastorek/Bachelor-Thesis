abstract environment to create CHSH framework

actions are expected in this format

ACTIONS = [q + axis + “0” for axis in ‘xyz’ for q in ‘ra’] ACTIONS = [q + axis + “0” for axis in ‘y’ for q in ‘r’] PLAYER = [‘a’, ‘b’] QUESTION = [‘0’, ‘1’]

ALL_POSSIBLE_ACTIONS = [[p + q + a] for p in PLAYER for q in QUESTION for a in ACTIONS] # place one gate at some place ALL_POSSIBLE_ACTIONS.append([“xxr0”])

# for 1 game with 2 EPR ALL_POSSIBLE_ACTIONS.append([“a0cxnot”]) ALL_POSSIBLE_ACTIONS.append([“b0cxnot”]) # # for xor paralel with 2EPR ALL_POSSIBLE_ACTIONS.append([“a0cxnotr”]) ALL_POSSIBLE_ACTIONS.append([“b0cxnotr”])
